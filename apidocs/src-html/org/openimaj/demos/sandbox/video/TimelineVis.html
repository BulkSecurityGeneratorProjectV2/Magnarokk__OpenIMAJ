042/**
043 * VisualisationImpl example of a timeline showing video, audio and other stuff.
044 *
045 * @author David Dupplaw (dpd@ecs.soton.ac.uk)
046 * @created 3 Jul 2012
047 * @version $Author$, $Revision$, $Date$
048 */
049public class TimelineVis
050{
051        /**
052         * Default main.
053         *
054         * @param args Command-line arguments
055         */
056        public static void main( final String[] args )
057        {
058                // Create a timeline
059                final Timeline t = new Timeline( 1500, 300 );
060
061                // Create a frame and display the timeline
062                t.showWindow( "Timeline" );
063
064                // Create a track with a video on it.
065                final TimelineTrack tt = t.addTrack( "Video 1" );
066                final ShotBoundaryVideoBarVisualisation sb = new ShotBoundaryVideoBarVisualisation(
067                                new XuggleVideo( new File( "video.m4v" ) ) );
068                t.addTimelineObject( tt, sb );
069                sb.setStartTimeMilliseconds( 2000 );
070                sb.processVideo();
