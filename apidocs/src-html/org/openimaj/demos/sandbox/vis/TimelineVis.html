044/**
045 * VisualisationImpl example of a timeline showing video, audio and other stuff.
046 *
047 * @author David Dupplaw (dpd@ecs.soton.ac.uk)
048 * @created 3 Jul 2012
049 * @version $Author$, $Revision$, $Date$
050 */
051public class TimelineVis
052{
053        /**
054         * Default main.
055         *
056         * @param args Command-line arguments
057         */
058        public static void main( final String[] args )
059        {
060                // Create a timeline
061                final Timeline t = new Timeline( 1200, 600 );
062
063                t.showWindow( "Timeline" );
064
065                // Create a track with a video on it.
066                final TimelineTrack tt = t.addTrack( "Video 1" );
067                final ShotBoundaryVideoBarVisualisation sb = new ShotBoundaryVideoBarVisualisation(
068                                new XuggleVideo( new File( "video.m4v" ) ) );
069                t.addTimelineObject( tt, sb );
070//              sb.setStartTimeMilliseconds( 2000 );
071                sb.processVideo();
072
073                final TimelineTrack tt2 = t.addTrack( "Audio 1" );
074                final AudioOverviewVisualisation aov = new AudioOverviewVisualisation(
075                                new XuggleAudio( new File("video.m4v") ) );
076                t.addTimelineObject( tt2, aov );

086//                              new XuggleVideo( new File( "videoplayback.mp4" ) ) );
