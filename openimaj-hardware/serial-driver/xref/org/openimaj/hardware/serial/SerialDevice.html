18  /**
19   * Serial device driver. Uses RXTX library underneath. The native parts of the
20   * RXTX library are published to the Maven repository as a JAR and are extracted
21   * and the java.library.path property is flushed and reset.
22   *
23   * @author David Dupplaw (dpd@ecs.soton.ac.uk)
24   *
25   * @created 12 Jul 2011
26   */
27  public class SerialDevice implements SerialDataListener
28  {
29  	/** The RXTX serial port we'll be reading from */
30  	private SerialPort serialPort = null;
31  
32  	/** Listeners for data events coming from the serial port */
33  	private List<SerialDataListener> listeners = new ArrayList<SerialDataListener>();
34  
35  	/** The regular expression used to split incoming data for the listeners */
36  	private String regex = "\n";
37  
38  	/** The input stream for the port */
39  	private InputStream inputStream = null;
40  
41  	/** The output stream for the port */
42  	private OutputStream outputStream = null;
43  
44  	/** The serial reader used to buffer and parse incoming data */
45  	private SerialReader serialReader = null;
46  
47  	/** The parser being used to parse incoming data */
48  	private RegExParser regexParser = null;

50  	/**
51  	 * Constructor that takes the port name to connect to and the rate at which
52  	 * to connect. The data rate will be set to 19,200 with 8 data bits, 1 stop
53  	 * bit and no parity.
54  	 *
55  	 * @param portName
56  	 *            The port name to connect to.
57  	 * @throws Exception
58  	 */
59  	public SerialDevice(String portName) throws Exception
60  	{
61  		this(portName, 4800, SerialPort.DATABITS_8, SerialPort.STOPBITS_1, SerialPort.PARITY_NONE);
62  	}

64  	/**
65  	 * Complete constructor that takes all the information required to connect
66  	 * to a port.
67  	 *
68  	 * @param portName
69  	 *            The port name to connect to.
70  	 * @param dataRate
71  	 *            The data rate to read from the port.
72  	 * @param dataBits
73  	 *            The number of data bits
74  	 * @param stopBits
75  	 *            The number of stop bits
76  	 * @param parity
77  	 *            The bit parity
78  	 * @throws Exception
79  	 */
80  	public SerialDevice(String portName, int dataRate, int dataBits, int stopBits, int parity)
81  			throws Exception
82  	{
83  		// Set the serial port information
84  		serialPort = new SerialPort(portName);
85  		serialPort.openPort();
86  		serialPort.setParams(dataRate, dataBits, stopBits, parity);
87  
88  		// Get the input and output streams from and to the serial port.
89  		outputStream = new OutputStream() {
90  			@Override
91  			public void write(int b) throws IOException {
92  				try {
93  					serialPort.writeByte((byte) b);
94  				} catch (final SerialPortException e) {
95  					throw new IOException(e);
96  				}
97  			}
98  
99  			@Override
100 			public void write(byte[] b) throws IOException {
101 				try {
102 					serialPort.writeBytes(b);
103 				} catch (final SerialPortException e) {
104 					throw new IOException(e);
105 				}
106 			}
107 		};
108 		inputStream = new InputStream() {
109 			@Override
110 			public int read() throws IOException {
111 				while (true) {
112 					try {
113 						if (!serialPort.isOpened())
114 							return -1;
115 
116 						return serialPort.readBytes(1, 100)[0];
117 					} catch (final SerialPortTimeoutException e) {
118 						// ignore and try again
119 					} catch (final SerialPortException e) {
120 						if (e.getMessage().contains("Port not opened"))
121 							return -1;
122 						throw new IOException(e);
123 					}
124 				}
125 			}
126 		};
127 
128 		// Set up our data listener
129 		regexParser = new RegExParser(regex);
130 		serialReader = new SerialReader(inputStream, regexParser);
131 		serialReader.addSerialDataListener(this);
132 		serialPort.addEventListener(serialReader, SerialPort.MASK_RXCHAR);
133 	}

135 	/**
136 	 * {@inheritDoc}
137 	 *
138 	 * @see java.lang.Object#finalize()
139 	 */
140 	@Override
141 	protected void finalize() throws Throwable
142 	{
143 		if (serialPort.isOpened()) {
144 			serialReader.close();
145 			serialPort.removeEventListener();
146 			serialPort.closePort();
147 		}
148 		super.finalize();
149 	}

156 	public void close() throws IOException
157 	{
158 		try {
159 			serialReader.close();
160 			serialPort.removeEventListener();
161 			serialPort.closePort();
162 		} catch (final SerialPortException e) {
163 			throw new IOException(e);
164 		}
165 	}
