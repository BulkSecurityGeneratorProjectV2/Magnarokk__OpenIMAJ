57  /**
58   * Fast, parallel implementation of the K-Means algorithm with support for
59   * bigger-than-memory data. Various flavors of K-Means are supported through the
60   * selection of different subclasses of {@link ObjectNearestNeighbours}; for
61   * example, exact K-Means can be achieved using an
62   * {@link ObjectNearestNeighboursExact}. The specific choice of
63   * nearest-neighbour object is controlled through the
64   * {@link NearestNeighboursFactory} provided to the {@link KMeansConfiguration}
65   * used to construct instances of this class. The choice of
66   * {@link ObjectNearestNeighbours} affects the speed of clustering; using
67   * approximate nearest-neighbours algorithms for the K-Means can produces
68   * comparable results to the exact KMeans algorithm in much shorter time. The
69   * choice and configuration of {@link ObjectNearestNeighbours} can also control
70   * the type of distance function being used in the clustering.
71   * <p>
72   * The algorithm is implemented as follows: Clustering is initiated using a
73   * {@link ByteKMeansInit} and is iterative. In each round, batches of samples
74   * are assigned to centroids in parallel. The centroid assignment is performed
75   * using the pre-configured {@link ObjectNearestNeighbours} instances created
76   * from the {@link KMeansConfiguration}. Once all samples are assigned new
77   * centroids are calculated and the next round started. Data point pushing is
78   * performed using the same techniques as center point assignment.
79   * <p>
80   * This implementation is able to deal with larger-than-memory datasets by
81   * streaming the samples from disk using an appropriate {@link DataSource}. The
82   * only requirement is that there is enough memory to hold all the centroids
83   * plus working memory for the batches of samples being assigned.
84   *
85   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
86   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
87   *
88   * @param <T>
89   *            Type of object being clustered
90   */
91  public class FeatureVectorKMeans<T extends FeatureVector> implements SpatialClusterer<FeatureVectorCentroidsResult<T>, T>

93  	private static class CentroidAssignmentJob<T extends FeatureVector> implements Callable<Boolean> {
94  		private final DataSource<T> ds;
95  		private final int startRow;
96  		private final int stopRow;
97  		private final ObjectNearestNeighbours<T> nno;
98  		private final double[][] centroids_accum;
99  		private final int[] counts;
100 
101 		public CentroidAssignmentJob(DataSource<T> ds, int startRow, int stopRow, ObjectNearestNeighbours<T> nno,
102 				double[][] centroids_accum, int[] counts)
103 		{
104 			this.ds = ds;
105 			this.startRow = startRow;
106 			this.stopRow = stopRow;
107 			this.nno = nno;
108 			this.centroids_accum = centroids_accum;
109 			this.counts = counts;
110 		}
111 
112 		@Override
113 		public Boolean call() {
114 			try {
115 				final int D = ds.getData(0).length();
116 
117 				final T[] points = ds.createTemporaryArray(stopRow - startRow);
118 				ds.getData(startRow, stopRow, points);
119 
120 				final int[] argmins = new int[points.length];
121 				final float[] mins = new float[points.length];
122 
123 				nno.searchNN(points, argmins, mins);
124 
125 				synchronized (centroids_accum) {
126 					for (int i = 0; i < points.length; ++i) {
127 						final int k = argmins[i];
128 						final double[] vector = points[i].asDoubleVector();
129 						for (int d = 0; d < D; ++d) {
130 							centroids_accum[k][d] += vector[d];
131 						}
132 						counts[k] += 1;
133 					}
134 				}
135 			} catch (final Exception e) {
136 				e.printStackTrace();
137 			}
138 			return true;
139 		}
140 	}

293 	/**
294 	 * Main clustering algorithm. A number of threads as specified are started
295 	 * each containing an assignment job and a reference to the same set of
296 	 * ObjectNearestNeighbours object (i.e. Exact or KDTree). Each thread is
297 	 * added to a job pool and started in parallel. A single accumulator is
298 	 * shared between all threads and locked on update.
299 	 *
300 	 * @param data
301 	 *            the data to be clustered
302 	 * @param centroids
303 	 *            the centroids to be found
304 	 */
305 	protected void cluster(DataSource<T> data, Result<T> result) throws Exception {
306 		final T[] centroids = result.centroids;
307 		final int K = centroids.length;
308 		final int D = centroids[0].length();
309 		final int N = data.size();
310 		final double[][] centroids_accum = new double[K][D];
311 		final int[] new_counts = new int[K];
312 
313 		final ExecutorService service = conf.threadpool;
314 
315 		for (int i = 0; i < conf.niters; i++) {
316 			System.err.println("Iteration " + i);
317 			for (int j = 0; j < K; j++)
318 				Arrays.fill(centroids_accum[j], 0);
319 			Arrays.fill(new_counts, 0);
320 
321 			final ObjectNearestNeighbours<T> nno = conf.factory.create(centroids);
322 
323 			final List<CentroidAssignmentJob<T>> jobs = new ArrayList<CentroidAssignmentJob<T>>();
324 			for (int bl = 0; bl < N; bl += conf.blockSize) {
325 				final int br = Math.min(bl + conf.blockSize, N);
326 				jobs.add(new CentroidAssignmentJob<T>(data, bl, br, nno, centroids_accum, new_counts));
327 			}
328 
329 			service.invokeAll(jobs);
330 
331 			for (int k = 0; k < K; ++k) {
332 				if (new_counts[k] == 0) {
333 					// If there's an empty cluster we replace it with a random
334 					// point.
335 					new_counts[k] = 1;
336 
337 					final T[] rnd = data.createTemporaryArray(1);
338 					data.getRandomRows(rnd);
339 
340 					final Cloner cloner = new Cloner();
341 					centroids[k] = cloner.deepClone(rnd[0]);
342 				} else {
343 					for (int d = 0; d < D; ++d) {
344 						centroids[k].setFromDouble(d, centroids_accum[k][d] / new_counts[k]);
345 					}
346 				}
347 			}
348 		}
349 	}
